```
% pandoc -t native
:::::::::: warning ::::::::::::
This is the warning!

1. list
2. another

::: {#myid .class key=val}
nested div
:::
:::::::::::::::::::::::::::::::
^D
[ Div
    ( "" , [ "warning" ] , [] )
    [ Para [ Str "This is the warning!" ]
    , OrderedList
        ( 1 , Decimal , Period )
        [ [ Plain [ Str "list" ] ] , [ Plain [ Str "another" ] ] ]
    , Div
        ( "myid" , [ "class" ] , [ ( "key" , "val" ) ] )
        [ Para [ Str "nested div" ] ]
    ]
]
```
